Header <boost/fixed_point/fixed_point.hpp>

namespace boost {
  namespace math {
    namespace fixed_point {
      template<const int range, const int resolution, 
               const round::round_type round_mode = round::nearest_odd, 
               const overflow::overflow_type overflow_mode = overflow::exception> 
        class negatable;
      namespace overflow {

        enum enum_overflow_type { impossible, undefined, modulus, saturate, 
                                  exception };

        typedef enum boost::math::fixed_point::overflow::enum_overflow_type overflow_type;
      }
      namespace round {

        enum enum_round_type { fastest, negative, truncated, positive, 
                               classic, nearest_even, nearest_odd };

        typedef enum boost::math::fixed_point::round::enum_round_type round_type;
      }
    }
  }
}namespace std {
  template<const int range, const int resolution, 
           const boost::math::fixed_point::round::round_type round_mode, 
           const boost::math::fixed_point::overflow::overflow_type overflow_mode> 
    class numeric_limits<boost::math::fixed_point::negatable< range, resolution, round_mode, overflow_mode >>;
}
